Theme Format

  Stream plugin framework

Overview

Eureka Streams allows for custom themes to be added by administrators. These themes are then visible for end-users to activate on the start page.

XSD

The following is the XML Schema Definition for a custom Theme file

<?xml version="1.0"?>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema">
    <xs:element name="Theme">
        <xs:complexType>
            <xs:all>
                <xs:element name="HeaderBackground">
                    <xs:complexType>
                        <xs:all>
                            <xs:element name="HeaderBackgroundColor" type="xs:string" />
                            <xs:element name="HeaderForegroundColor" type="xs:string" />
                            <xs:element name="HeaderBackgroundImage">
                                <xs:complexType>
                                    <xs:simpleContent>
                                        <xs:extension base="xs:string">
                                            <xs:attribute name="Center" type="xs:boolean"
                                                default="false" />
                                        </xs:extension>
                                    </xs:simpleContent>
                                </xs:complexType>
                            </xs:element>
                        </xs:all>
                    </xs:complexType>
                </xs:element>
                <xs:element name="Title" type="xs:string" />
                <xs:element name="Description" type="xs:string" />
                <xs:element name="AuthorName" type="xs:string" />
                <xs:element name="AuthorEmail" type="xs:string" />
                <xs:element name="ActiveTabBorderColor" type="xs:string" />
                <xs:element name="ActiveTabForegroundColor" type="xs:string" />
                <xs:element name="InactiveTabBackgroundColor" type="xs:string" />
                <xs:element name="InactiveTabBorderColor" type="xs:string" />
                <xs:element name="InactiveTabForegroundColor" type="xs:string" />
                <xs:element name="GadgetTitleBackgroundColor" type="xs:string" />
                <xs:element name="GadgetTitleForegroundColor" type="xs:string" />
                <xs:element name="GadgetBorderColor" type="xs:string" />
                <xs:element name="PageBackground">
                    <xs:complexType>
                        <xs:all>
                            <xs:element name="PageBackgroundColor" type="xs:string" />
                            <xs:element name="PageBackgroundImage">
                                <xs:complexType>
                                    <xs:simpleContent>
                                        <xs:extension base="xs:string">
                                            <xs:attribute name="RepeatX" type="xs:boolean"
                                                default="false" />
                                            <xs:attribute name="RepeatY" type="xs:boolean"
                                                default="false" />
                                        </xs:extension>
                                    </xs:simpleContent>
                                </xs:complexType>
                            </xs:element>
                        </xs:all>
                    </xs:complexType>
                </xs:element>
            </xs:all>
        </xs:complexType>
    </xs:element>
</xs:schema>

Sample XML

The following is a sample theme conforming to the XSD above:

<?xml version="1.0" encoding="UTF-8"?>
<Theme>
    <Title>My Theme</Title>
    <Description>A theme for Eureka Streams</Description>
    <AuthorName>Eureka Streams</AuthorName>
    <AuthorEmail>support@eurekastreams.org</AuthorEmail>
    <HeaderBackground>
        <HeaderBackgroundColor>white</HeaderBackgroundColor>
        <HeaderForegroundColor>white</HeaderForegroundColor>
        <HeaderBackgroundImage>/style/images/Start_page.jpg</HeaderBackgroundImage>
    </HeaderBackground>
    <ActiveTabBorderColor>#A4A5A9</ActiveTabBorderColor>
    <ActiveTabForegroundColor>#313007</ActiveTabForegroundColor>
    <InactiveTabBackgroundColor>#505657</InactiveTabBackgroundColor>
    <InactiveTabBorderColor>#2E3131</InactiveTabBorderColor>
    <InactiveTabForegroundColor>white</InactiveTabForegroundColor>
    <GadgetTitleBackgroundColor>#2E4523</GadgetTitleBackgroundColor>
    <GadgetTitleForegroundColor>white</GadgetTitleForegroundColor>
    <GadgetBorderColor>#313007</GadgetBorderColor>
    <PageBackground>
        <PageBackgroundColor>#E6E1D0</PageBackgroundColor>
        <PageBackgroundImage/>
    </PageBackground>
</Theme>
